mysql角色用户

CREATE ROLE IF NOT EXISTS 'test_opr', 'test_data';
GRANT ALL ON TCC.* TO test_data;
GRANT SELECT ON TCC.* TO test_opr;

CREATE USER 'test'@'%' IDENTIFIED BY 'test';

GRANT test_data TO 'test'@'%';

--验证
SHOW GRANTS FOR test@'%';
